## Endpoint

```http
GET https://contentstats.io/api/v1/videos
```

Retrieve all videos you're tracking, with optional filters for platform and status.

## Authentication

<ParamField header="X-API-Key" type="string" required>
  Your API key from the [dashboard](https://contentstats.io/dashboard/api-keys)
</ParamField>

## Query Parameters

<ParamField query="platform" type="string">
  Filter by platform
  
  **Options:** `tiktok`, `youtube`, `instagram`, `twitter`
  
  **Example:** `?platform=tiktok`
</ParamField>

<ParamField query="status" type="string">
  Filter by tracking status
  
  **Options:** `monitoring`, `completed`, `paused`, `error`, `cancelled`
  
  **Example:** `?status=monitoring`
</ParamField>

<ParamField query="limit" type="integer">
  Number of results to return (1-100)
  
  **Default:** 50
  
  **Example:** `?limit=100`
</ParamField>

## Response

<ResponseField name="videos" type="array">
  Array of video objects
  
  <Expandable title="Video Object">
    <ResponseField name="id" type="string">
      Unique video identifier
    </ResponseField>
    
    <ResponseField name="job_id" type="string">
      External job ID
    </ResponseField>
    
    <ResponseField name="video_link" type="string">
      Video URL
    </ResponseField>
    
    <ResponseField name="platform" type="string">
      Platform: `tiktok`, `youtube`, `instagram`, `twitter`
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Current status
    </ResponseField>
    
    <ResponseField name="monitoring_started" type="string">
      ISO 8601 timestamp
    </ResponseField>
    
    <ResponseField name="monitoring_duration_days" type="integer">
      Total tracking duration
    </ResponseField>
    
    <ResponseField name="days_remaining" type="number">
      Days left in tracking
    </ResponseField>
    
    <ResponseField name="current_metadata" type="object">
      Latest video metadata
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      When tracking started
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer">
  Number of videos in response
</ResponseField>

<ResponseField name="limit" type="integer">
  Applied limit
</ResponseField>

## Code Examples

<CodeGroup>
```bash cURL - All Videos
curl "https://contentstats.io/api/v1/videos" \
  -H "X-API-Key: cs_live_YOUR_KEY"
```

```bash cURL - Filter by Platform
curl "https://contentstats.io/api/v1/videos?platform=tiktok&status=monitoring" \
  -H "X-API-Key: cs_live_YOUR_KEY"
```

```javascript JavaScript
// Get all TikTok videos currently being monitored
const response = await fetch(
  'https://contentstats.io/api/v1/videos?platform=tiktok&status=monitoring',
  {
    headers: {
      'X-API-Key': process.env.CONTENTSTATS_API_KEY
    }
  }
);

const data = await response.json();

console.log(`Tracking ${data.total} TikTok videos`);
data.videos.forEach(video => {
  console.log(`${video.video_link}: ${video.days_remaining} days left`);
});
```

```python Python
import requests
import os

# Get all videos
response = requests.get(
    'https://contentstats.io/api/v1/videos',
    headers={'X-API-Key': os.environ['CONTENTSTATS_API_KEY']}
)

data = response.json()

# Filter by status
active = [
    v for v in data['videos']
    if v['status'] == 'monitoring'
]

print(f"Found {len(active)} active videos")
```

```php PHP
<?php
$curl = curl_init();

curl_setopt_array($curl, [
    CURLOPT_URL => 'https://contentstats.io/api/v1/videos?platform=youtube&limit=100',

$response = curl_exec($curl);
$data = json_decode($response, true);

echo "Total YouTube videos: " . $data['total'];
?>
```

## Use Cases

### Dashboard - Active Videos

```javascript
async function getActiveVideos() {
  const response = await fetch(
    'https://contentstats.io/api/v1/videos?status=monitoring',
    { headers: { 'X-API-Key': apiKey } }
  );
  
  const data = await response.json();
  
  return data.videos.map(v => ({
    id: v.id,
    platform: v.platform,
    url: v.video_link,
    daysLeft: v.days_remaining
  }));
}
```

### Platform-Specific Analytics

```javascript
async function getPlatformStats(platform) {
  const response = await fetch(
    `https://contentstats.io/api/v1/videos?platform=${platform}`,
    { headers: { 'X-API-Key': apiKey } }
  );
  
  const data = await response.json();
  
  const stats = {
    total: data.total,
    active: data.videos.filter(v => v.status === 'monitoring').length,
    totalDaysTracking: data.videos.reduce((sum, v) => sum + v.monitoring_duration_days, 0)
  };
  
  return stats;
}

const tiktokStats = await getPlatformStats('tiktok');
console.log('TikTok:', tiktokStats);
```

## Combining Filters

```bash
# Get all completed TikTok videos
curl "https://contentstats.io/api/v1/videos?platform=tiktok&status=completed&limit=100" \
  -H "X-API-Key: YOUR_KEY"

# Get first 10 monitoring YouTube videos
curl "https://contentstats.io/api/v1/videos?platform=youtube&status=monitoring&limit=10" \
  -H "X-API-Key: YOUR_KEY"
```

## Pagination

The API returns up to 100 videos per request. For more videos:

```javascript
async function getAllVideos() {
  const allVideos = [];
  let hasMore = true;
  let limit = 100;
  
  // Note: Current API doesn't support offset
  // This example shows the pattern for when pagination is added
  
  const response = await fetch(
    `https://contentstats.io/api/v1/videos?limit=${limit}`,
    { headers: { 'X-API-Key': apiKey } }
  );
  
  const data = await response.json();
  allVideos.push(...data.videos);
  
  return allVideos;
}
```

## Best Practices

<AccordionGroup>
  <Accordion icon="filter" title="Use Filters to Reduce Response Size">
    Only request what you need:
    
    ```javascript
    // ❌ Don't: Get all videos when you only need TikTok
    const all = await fetch('/api/v1/videos');
    const tiktok = all.videos.filter(v => v.platform === 'tiktok');
    
    // ✅ Do: Filter server-side
    const tiktok = await fetch('/api/v1/videos?platform=tiktok');
    ```
  </Accordion>
  
  <Accordion icon="database" title="Cache Responses">
    Videos don't change frequently:
    
    ```javascript
    const cache = new Map();
    const TTL = 5 * 60 * 1000; // 5 minutes
    
    async function getCachedVideos(platform) {
      const key = `videos_${platform}`;
      const cached = cache.get(key);
      
      if (cached && Date.now() - cached.timestamp < TTL) {
        return cached.data;
      }
      
      const response = await fetch(
        `https://contentstats.io/api/v1/videos?platform=${platform}`,
        { headers: { 'X-API-Key': apiKey } }
      );
      
      const data = await response.json();
      cache.set(key, { data, timestamp: Date.now() });
      
      return data;
    }
    ```
  </Accordion>
  
  <Accordion icon="clock" title="Poll Status Changes">
    Monitor video status changes:
    
    ```javascript
    async function monitorStatusChanges() {
      const previous = new Map();
      
      setInterval(async () => {
        const response = await fetch('/api/v1/videos?status=monitoring');
        const data = await response.json();
        
        data.videos.forEach(video => {
          const prev = previous.get(video.id);
          
          if (prev && prev.days_remaining !== video.days_remaining) {
            console.log(`${video.id}: ${video.days_remaining} days left`);
          }
          
          previous.set(video.id, video);
        });
      }, 60 * 60 * 1000); // Check hourly
    }
    ```
  </Accordion>
</AccordionGroup>
